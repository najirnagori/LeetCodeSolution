// LeetCode Problem: Best time to buy stockcs

// Solution:

func maxProfit(_ prices: [Int]) -> Int {
    var minPrice = Int.max
    var maxProfit = 0
    
    for price in prices {
        if price < minPrice {
            minPrice = price
        }
        
        let profit = price - minPrice
        
        if profit > maxProfit {
            maxProfit = profit
        }
    }
    return  maxProfit
}
