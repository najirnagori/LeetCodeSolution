// majority element: We can done this by using sorting but, there is one more optimal solution for this which is moore voting algorithm. and we will done from this

func majorityElement(_ nums: [Int]) -> Int {
    var result = 0
    var frequency = 0
    
    for num in nums {
        if frequency == 0 {
            result = num
        }
        
        if num == result {
            frequency += 1
        } else {
            frequency -= 1
        }
    }
    return result
}
